import streamlit as st
import pandas as pd
import joblib
import numpy as np

# 1. Load the "Brain" (Model and Scaler)
model = joblib.load('diabetes_model.pkl')
scaler = joblib.load('scaler.pkl')

st.title("🩺 Diabetes Risk Predictor")
st.write("Enter your health metrics below to see your risk level.")

# 2. User Input Form
col1, col2 = st.columns(2)

with col1:
    glucose = st.number_input("Glucose Level", min_value=0, max_value=300, value=100)
    bmi = st.number_input("BMI", min_value=0.0, max_value=70.0, value=25.0)
    age = st.number_input("Age", min_value=1, max_value=120, value=30)
    bp = st.number_input("Blood Pressure", min_value=0, max_value=200, value=80)

with col2:
    insulin = st.number_input("Insulin Level", min_value=0, max_value=900, value=30)
    skin = st.number_input("Skin Thickness", min_value=0, max_value=100, value=20)
    pedigree = st.number_input("Diabetes Pedigree Function", min_value=0.0, max_value=3.0, value=0.5)
    pregnancies = st.number_input("Pregnancies", min_value=0, max_value=20, value=0)

# 3. Prediction Logic
if st.button("Predict Results"):
    # Arrange data in the same order as training
    user_data = np.array([[pregnancies, glucose, bp, skin, insulin, bmi, pedigree, age]])
    
    # SCALE the data (Important for accuracy!)
    user_data_scaled = scaler.transform(user_data)
    
    prediction = model.predict(user_data_scaled)
    probability = model.predict_proba(user_data_scaled)[0][1]

    if prediction[0] == 1:
        st.error(f"High Risk Detected. Probability: {probability:.2%}")
    else:
        st.success(f"Low Risk Detected. Probability: {probability:.2%}")
